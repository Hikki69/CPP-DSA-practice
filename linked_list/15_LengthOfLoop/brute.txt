int lengthOfLoop(Node* head){
    Node* temp = head;
    map<Node*,int>mpp;
    int timer = 1;
    while(temp!=NULL){
        if(mpp.find(temp) != mpp.end()){
            int val = mpp[temp];
            return (timer-val);
        }
        mpp[temp] = timer;
        timer++;
        temp = temp->next;
    }
    return 0;
}


// time complexity 
// TC = O(n x 2 x TC(map))

// space complexity
// SC = O(n)
