int lengthOfLoop(Node* head){
    Node* slow = head;
    Node* fast = head;
    while(fast!=NULL && fast->next!=NULL){
        slow = slow->next;
        fast = fast->next->next;
        if(slow == fast)return findlength(slow,fast);
    }
    return 0;
}

int findlength(Node* slow, Node* fast){
    int cnt =1;
    fast = fast->next;
    while(slow!=fast){
        cnt++;
        fast = fast->next;
    }
    return cnt;
}


// time complexity 
// TC = O(n)

// space complexity
// SC = O(1)
