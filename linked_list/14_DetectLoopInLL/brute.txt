bool loopInL(Node* head){
    map<Node*,int>mpp;
    Node* temp=head;
    while(temp!=NULL){
        if(mpp.find(temp)) return true;
        mpp[temp] = 1;
        temp = temp->next;
    }
    return false;
}

// time complexity 
// TC = O(N + 2 x Tc(map))

// space complexity
// SC = O(N)
